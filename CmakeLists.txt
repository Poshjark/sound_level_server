cmake_minimum_required(VERSION 3.0)

project(sound_level_server LANGUAGES CXX)

find_package(Boost 1.78.0)
INCLUDE_DIRECTORIES(${Boost_INCLUDE_DIR})

set(SOURCE_EXE testServer.cpp)

set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin)

set(VOULME_HANDLER_SOURCE_LIB volume_handler.cpp volume_handler.h)

add_library(volume_handler ${VOULME_HANDLER_SOURCE_LIB})

set(COMMAND_HANDLER_SOURCE_LIB command_handler.cpp command_handler.h)

add_library(command_handler ${COMMAND_HANDLER_SOURCE_LIB})

add_executable(main ${SOURCE_EXE})

target_link_libraries(main volume_handler command_handler ${Boost_LIBRARIES})
